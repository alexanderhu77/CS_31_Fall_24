The program prints out one "#" on the first line, two "#" on the second, etc. up until it reaches the number of lines the user enters.

#include <iostream>
using namespace std;

int main()
{
	int side;

	cout << "Enter a number: ";
	cin >> side;
	for (int i = 0; i < side; i++){
		int j = i;
		while (j >= 0) {
			cout << "#";
			j--;
		}
		cout << "\n";
	}

}
